 14 /************************* IO **************************/
 15 DigitalOut E_1(D8,0);               // Control de electrovalvula 1
 16 DigitalOut E_2(D9,0);               // Control de electrovalvula 2
 17 DigitalOut E_3(D10,0);              // Control de electrovalvula 3
 18 DigitalOut E_4(D12,0);              // Control de electrovalvula 4
 19 PwmOut motor_plus(PB_8);            // Variables para el motor
 20 PwmOut motor_cero(PB_9);        
 21 QEI encoder(PD_6,PD_5,NC,4576,QEI::X2_ENCODING);
 22 
 23 /******************* INTERRUPCIONES ********************/
 24 Timeout t;
 25 InterruptIn button(USER_BUTTON);    // Cancelar operación
 26 
 27 /********************* VARIABLES ***********************/
 28 int tempUser1 = 1;                  // Tiempo de electrocoagulacion (tanque 1)
 29 int tempUser2 = 1;                  // Tiempo de reposo de la vinaza en el tanque 2
 30 int tempUser3 = 1;                  // Tiempo de vaciado del agua del tanque 2
 31 int tempUser4 = 1;                  // Tiempo de vaciado de residuos sólidos del tanque 2
 32 int pwm = 50;                       // Duty cycle del motor
 33 
 34 int tiempoLlenado = 5;              // Tiempo de llenado del tanque 1
 35 int timeScale = 60;                 // 1 -> segundos, 60 -> minutos. Afecta los tiempos tempUser1 y tempUser2
 36  
 37 int a = 0;
 38 int b = 0;
 39 int c = 0;
 40 int running = 0;
 41 int finished = 0;
 42 int canceled = 0;
 43 int campo = 1;
 44 
 45 /******************** TECLADO Y LCD ***********************/
 46 // Definición de los IO del teclado y LCD
 47 Serial pc(USBTX, USBRX); //rs, e, d4, d5, d6, d7
 48 TextLCD lcd(PC_8,PC_9,PC_10,PC_11,PC_12,PD_2);
 49 Keypad keypad(D3, D2, D1, D0, D7, D6, D5, D4);
 50 
 51 /********************* FUNCIONES **************************/
 52 void printLCD() {                               // Función para la impresión de la pantalla LCD
 53     while(1) {
 54         lcd.cls();
 55         lcd.locate(0,0);
 56         lcd.printf("t1=%d",tempUser1);
 57         lcd.locate(6,0);
 58         lcd.printf("t2=%d",tempUser2);
 59         lcd.locate(0,1);
 60         lcd.printf("t3=%d",tempUser3);
 61         lcd.locate(6,1);
 62         lcd.printf("t4=%d",tempUser4);
 63         lcd.locate(12,0);
 64         lcd.printf("PWM");
 65         lcd.locate(12,1);
 66         lcd.printf("%d",pwm);
 67         wait_ms(150);
 68         switch(campo) {                         // La variable pendiente de modificar titila con un periodo de 300 ms
 69             case  1:
 70                 lcd.locate(0,0);
 71                 lcd.printf("t1=   ",tempUser1);
 72                 break;
 73             case  2:
 74                 lcd.locate(6,0);
 75                 lcd.printf("t2=   ",tempUser2);
 76                 break;
 77             case  3:
 78                 lcd.locate(0,1);
 79                 lcd.printf("t3=   ",tempUser3);
 80                 break;
 81             case  4:
 82                 lcd.locate(6,1);
 83                 lcd.printf("t4=   ",tempUser4);
 84                 break;
 85             case  5:
 86                 lcd.locate(12,1);
 87                 lcd.printf("   ");
 88                 break;
 89             default: ;
 90                 break;
 91         }
 92         wait_ms(150);
 93     }
 94 }

 96 int getNum(){                                   // Función que retorna un valor entero por cada tecla presionada
 97     while(1){
 98         char key = keypad.getKey();
 99         switch(key){
100             case '0': return 0;
101             case '1': return 1;
102             case '2': return 2;
103             case '3': return 3;
104             case '4': return 4;
105             case '5': return 5;
106             case '6': return 6;
107             case '7': return 7;
108             case '8': return 8;
109             case '9': return 9;
110             case 'A':
111                 campo++;
112                 if(campo == 6) campo = 1;
113                 return -2;
114           //case 'B': return -3;
115           //case 'C': return -4;
116             case 'D': return -5;
117             default: ; break;
118         }
119     }
120 }
121 void reiniciar(){                               // Función para apagar las válvulas y el motor
122     E_1 = 0;
123     E_2 = 0;
124     E_3 = 0;
125     E_4 = 0;
126     
127     motor_plus.write(0.0f);                     
128     motor_cero.write(0.0f);
129 }

132 void init(){                                    // Función de iniciación del programa
133     
134     keypad.enablePullUp();
135     wait(1);
136     lcd.locate(0,0);
137     lcd.printf("  Control para\n");
138     lcd.locate(0,1);
139     lcd.printf("electrocoagular\n");
140     wait(1.5);
141     
142     return;
143 }
144 
145 void setUserTimes(){                            // Algoritmo para la definición de las variables tempUser1, tempUser2,
146     while(1){                                   // tempUser3, tempUser4 y PWM
147     setvalue:                                   // Presionar la tecla 'A' cambia el campo a modificar
148         while(keypad.getKey()!='*'){}           // Presionar la tecla 'D' inicia el proceso de electrocuoagulación
149         a = 100*getNum();
150             if(a == -200) goto setvalue;
151             if(a == -500) break;
152         while(keypad.getKey()!='*'){}
153         b = 10*getNum();
154             if(b == -20) goto setvalue;
155             if(b == -50) break;
156         while(keypad.getKey()!='*'){}
157         c = getNum();
158             if(c == -2) goto setvalue;
159             if(c == -5) break;
160 
161         switch(campo){
162             case 1:
163                 tempUser1 = a+b+c;
164                 break;
165             case 2:
166                 tempUser2 = a+b+c;
167                 break;
168             case 3:
169                 tempUser3 = a+b+c;
170                 break;
171             case 4:
172                 tempUser4 = a+b+c;
173                 break;
174             case 5:
175                 pwm=a+b+c;
176                 if(pwm < 5)   pwm = 5;
177                 if(pwm > 100) pwm = 100;
178                 break;
179             default: ; break;
180         }
181     }
182     return;
183 }

185 /********************* CONTROL INTERRUPCIONES (TICKER) *************************/
186 void vaciadoSolido(){                           // Fin del procedimiento
187     E_4 = 0;                                    // Se cierra la EV 4 y se alza la bandera "finished"
188     finished = 1;
189 }
190 
191 void vaciadoLiquido(){                          // Inicio del vaciado del sólido
192     E_3 = 0;                                    // Se cierra la EV 3 y se abre la EV 4
193     E_4 = 1;
194     t.attach(&vaciadoSolido, tempUser4);
195 }
196 
197 void reposo(){                                  // Inicio del vaciado del líquido
198     E_3 = 1;                                    // Se abre la EV 3
199     t.attach(&vaciadoLiquido, tempUser3);
200 }
201 
202 void vaciado1(){                                // Inicio de la etapa de reposo en el tanque 2
203     E_2 = 0;                                    // Se cierra la EV2
204     t.attach(&reposo, timeScale*tempUser2);
205 }
206 
207 void electro(){                                 // Inicio del vaciado del primer tanque al segundo
208     motor_plus.write(0.0f);                     // Se apaga el motor y se abre la EV 2
209     motor_cero.write(0.0f);
210     E_2 = 1;
211     t.attach(&vaciado1, tiempoLlenado);
212 }
213 
214 void tiempoValvula1(){                          // Inicio del proceso de electrocoagulación
215     E_1 = 0;                                    // Cierre de la EV 1 y encendido del motor 
216     motor_plus.write((float)pwm*0.01f);
217 
218     t.attach(&electro, timeScale*tempUser1);
219 }
220 
221 void cancel_op(){                               // Llamado por la interrupción externa "button"
222     canceled = 1;                               // Cancela la operación en curso
223     t.detach();
224 }
225 
226 /**************************** MAIN ********************************/
227 
228 int main(){
229     motor_plus.period(0.015f);
230     motor_cero.write(0.0f);
231 
232     button.rise(&cancel_op);                    // Habilita (masks) la interrupción externa para cancelar la operación
233     while(1){
234         if(running == 0){                       // Llamado cada vez que se reinicia el sistema
235             running = 1;
236             init();                             // Función de iniciación del sistema
237             Thread lcdprint;                    // Inicia un hilo independiente para la impresión de la LCD
238             lcdprint.start(printLCD);    
239 
240             campo = 1;
241             setUserTimes();                     // Función en loop para definir las variables del programa
242 
243             campo = 0;
244             finished = 0;
245             canceled = 0;
246 
247             E_1 = 1;
248             t.attach(&tiempoValvula1, tiempoLlenado); // Inicio del proceso de electrocuagulación
249         }
250         if(finished == 1){                      // Llamado si el proceso termina exitosamente
251             reiniciar();
252             lcd.cls();
253             lcd.locate(0,0);
254             lcd.printf("Finalizado\n");
255             running = 0;
256             wait(2);
257         }
258         if(canceled == 1){                      // Llamado si el proceso es cancelado
259             reiniciar();
260             lcd.cls();
261             lcd.locate(0,0);
262             lcd.printf("Operacion\n");
263             lcd.locate(0,1);
264             lcd.printf("cancelada\n");
265             running = 0;
266             wait(2);
267         }
268     }
269 }
